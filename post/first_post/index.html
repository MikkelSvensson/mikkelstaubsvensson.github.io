max=None
min=None
while True:
   number = input("Enter a number: ")
   if number.upper() == "DONE":
      print(f'Max: {max} Min: {min}')
      break
   try:
      number=float(number)
   except(ValueError):
      print("Invalid input")
      continue
   if max is None or number > max:
      max = number
   elif min is None or number < min:
      min = number
